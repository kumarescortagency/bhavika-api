const express = require('express');
const app = express();

app.get('/', (req, res) => {
  res.send('API is Live 🚀');
});

app.get('/users', (req, res) => {
  res.json([
    { id: 1, name: "Puja" },
    { id: 2, name: "Rahul" }
  ]);
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log("Server running"));
